begin
  var x : int;
  var result : int;

  x := 100;            // globální x

  function shadow(x : int) : int;
  begin
    var temp : int;
    temp := x + 1;     
    return temp;
  end

  result := shadow(5); 
  print(x);            // vytiskne globální x = 100
  print(result);       // vytiskne 6

  return result;
end
