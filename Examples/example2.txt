begin
  var result : int;
  
  // Funkce jako parametr - UPRAVENO: pridana uplna signatura
  function apply(f : function(int) : int, x : int) : int;
  begin
    return f(x);
  end

  function square(n : int) : int;
  begin
    return n * n;
  end

  result := apply(square, 5);
  
  return result;
end
